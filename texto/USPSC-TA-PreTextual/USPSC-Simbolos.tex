
\begin{simbolos}

  \item[$\alpha$] Letra Grega alfa minúscula; nível de significância estatística.
  \item[$\beta$] Letra Grega beta minúscula; coeficiente de regressão (inclinação da reta/hiperplano).
  \item[$\beta_0$] Intercepto da regressão linear.
  \item[$\epsilon$] Letra Grega épsilon minúscula; termo de erro aleatório na regressão.
  \item[$\mu$] Letra Grega mu minúscula; média.
  \item[$\sigma$] Letra Grega sigma minúscula; desvio padrão.
  \item[$\le$] Menor ou igual
  \item[$Md$] Mediana
  \item[$F$] Estatística de teste da Análise de Variância (ANOVA)
  \item[$H_0$] Hipótese Nula
  \item[$n$] Tamanho da amostra ou número de observações
  \item[$p$-valor] Probabilidade de significância (valor $p$)
  \item[$Q1$] Primeiro Quartil (25\%)
  \item[$Q3$] Terceiro Quartil (75\%)
  \item[$R^2$] Coeficiente de determinação

\end{simbolos}
